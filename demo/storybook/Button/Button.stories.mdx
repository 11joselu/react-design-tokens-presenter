<!-- Read Button tokens -->

export const scssReq = require.context(
  `!!raw-loader!../`,
  true,
  /.ButtonTokens.scss$/
);

<!-- Create array of files -->

export const files = scssReq.keys();

## Colors

<div>
  {files
    .filter((filename) => filename.includes('Button'))
    // Read file contents
    .map((filename) => scssReq(filename).default)
    // parse the contents with a tokenizer
    .map(scssParser)
    .flat()
    .map((result) => {
      return (
        <div>
          <Color variableName={result.declaration} color={result.value}></Color>
          <br></br>
        </div>
      );
    })}
</div>
