# Color

Convey meaning through color.

export const scssReq = require.context(`!!raw-loader!./`, true, /tokens.scss$/);

export const files = scssReq.keys();

## Colors

<div>
  {files
    .map((filename) => scssReq(filename).default)
    .map(scssParser)
    .flat()
    .map((result, index) => {
      return (
        <div key={index}>
          <Color name={result.declaration} value={result.value}></Color>
          <br></br>
        </div>
      );
    })}
</div>
